package oops;
enum Gender{M,F}
public class Person 
{
	String firstName;
	String lastName;
	short age;
	double weight;
	short phone;

	Person()
	{
		firstName = "Divya";
		lastName = "Bharathi";
		age = 20;
		weight = 85.55;
		phone = 12345;
	}

	Person(String firstName, String lastName, short age, double weight, short phone)
	{
		this.firstName = firstName;
		this.lastName = lastName;
		this.age = age;
		this.weight = weight;
		this.phone = phone;
	}

	public String getFirstName() {
		return firstName;
	}

	public void setFirstName(String firstName) {
		this.firstName = firstName;
	}

	public String getLastName() {
		return lastName;
	}

	public void setLastName(String lastName) {
		this.lastName = lastName;
	}


	public short getAge() {
		return age;
	}

	public void setAge(short age) {
		this.age = age;
	}

	public double getWeight() {
		return weight;
	}

	public void setWeight(double weight) {
		this.weight = weight;
	}

	public short getPhone() {
		return phone;
	}

	public void setPhone(short phone) {
		this.phone = phone;
	}
}
